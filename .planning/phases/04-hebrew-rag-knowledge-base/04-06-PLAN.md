       ```python
       """RAG evaluation harness — recall@5 measurement (RAG-05).

       Mirrors scripts/eval_llm.py (Plan 03-06) argparse pattern. Two flow modes:

         --seed-only:  Ingest fixtures/rag/seed_kb/*.md via /api/kb/ingest-text
         --query-only: Run fixtures/rag/eval_questions.jsonl through /api/kb/query
         --full:       Both, in order

       Backend can be live ASGI (--backend-url, default http://localhost:8080) or
       in-process FastAPI TestClient (--testclient flag — CI-friendly, no
       running ChromaDB or Ollama needed in mock mode; live mode requires both
       up because BgeM3Embedder.create_and_verify is called at lifespan).

       Recall@5 formula:
       - For each question with gold_filename != null:
           hit = (any chunk in top-5 has source.filename == gold_filename
                  AND gold_chunk_match in chunk.text)
       - For each question with gold_filename == null:
           hit = (zero chunks returned — refusal correctness)
       - recall_at_5 = sum(hits) / n_questions

       Exit codes:
           0 success
           1 backend unreachable (HTTP error from /api/kb/health pre-check)
           2 recall@5 below 0.5 floor (Phase 7 prompt-tuner attention bar)

       STT-isolation contract: this harness imports receptra.rag.* + receptra.llm.schema
       only. NEVER receptra.stt + faster_whisper + silero_vad + torch + onnxruntime
       + ctranslate2 + av. test_recall_live.py companion test enforces via fresh
       subprocess sys.modules introspection.
       """
       from __future__ import annotations

       import argparse
       import json
       import sys
       from pathlib import Path
       from typing import Any

       import httpx

       _DEFAULT_FIXTURES_DIR = Path(__file__).resolve().parent.parent / "fixtures" / "rag"
       _DEFAULT_SEED_DIR = _DEFAULT_FIXTURES_DIR / "seed_kb"
       _DEFAULT_QUESTIONS = _DEFAULT_FIXTURES_DIR / "eval_questions.jsonl"


       def _load_questions(path: Path) -> list[dict[str, Any]]:
           with path.open("r", encoding="utf-8") as f:
               return [json.loads(line) for line in f if line.strip()]


       def _ingest_via_http(client: httpx.Client, seed_dir: Path) -> dict[str, int]:
           seeded = 0
           chunks_added = 0
           chunks_replaced = 0
           for md in sorted(seed_dir.glob("*.md")):
               content = md.read_text(encoding="utf-8")
               resp = client.post(
                   "/api/kb/ingest-text",
                   json={"filename": md.name, "content": content},
                   timeout=60.0,
               )
               resp.raise_for_status()
               body = resp.json()
               seeded += 1
               chunks_added += body["chunks_added"]
               chunks_replaced += body["chunks_replaced"]
           return {
               "seeded": seeded,
               "chunks_added": chunks_added,
               "chunks_replaced": chunks_replaced,
           }


       def _query_and_score(
           client: httpx.Client,
           questions: list[dict[str, Any]],
           top_k: int = 5,
       ) -> dict[str, Any]:
           details: list[dict[str, Any]] = []
           hits = 0
           refusal_correct = 0
           for q in questions:
               resp = client.post(
                   "/api/kb/query",
                   json={"query": q["question"], "top_k": top_k},
                   timeout=30.0,
               )
               resp.raise_for_status()
               results = resp.json()  # list[{id, text, source}]
               retrieved_filenames = [
                   (r.get("source") or {}).get("filename", "") for r in results
               ]
               row: dict[str, Any] = {
                   "question": q["question"],
                   "gold_filename": q["gold_filename"],
                   "gold_chunk_match": q.get("gold_chunk_match"),
                   "retrieved_filenames": retrieved_filenames,
                   "n_results": len(results),
               }
               if q["gold_filename"] is None:
                   # Refusal correctness: hit iff zero results.
                   hit = len(results) == 0
                   if hit:
                       refusal_correct += 1
               else:
                   match = q["gold_chunk_match"] or ""
                   hit = any(
                       (r.get("source") or {}).get("filename") == q["gold_filename"]
                       and match in r.get("text", "")
                       for r in results
                   )
               row["hit"] = hit
               if hit:
                   hits += 1
               details.append(row)

           n_with_gold = sum(1 for q in questions if q["gold_filename"] is not None)
           n_no_gold = len(questions) - n_with_gold
           return {
               "recall_at_5": round(hits / max(len(questions), 1), 4),
               "n_questions": len(questions),
               "n_with_gold": n_with_gold,
               "n_no_gold": n_no_gold,
               "refusal_correct": refusal_correct,
               "hits": hits,
               "details": details,
           }


       def _make_client(backend_url: str, testclient: bool) -> httpx.Client:
           if testclient:
               from fastapi.testclient import TestClient
               from receptra.main import app  # NOTE: imports settings + lifespan stack
               return TestClient(app)  # type: ignore[return-value]
           return httpx.Client(base_url=backend_url, timeout=60.0)


       def _health_check(client: httpx.Client) -> bool:
           try:
               resp = client.get("/api/kb/health", timeout=10.0)
               return resp.status_code == 200
           except httpx.HTTPError:
               return False


       def main() -> int:
           parser = argparse.ArgumentParser(
               prog="eval_rag", description="Receptra RAG evaluation harness (recall@5)",
           )
           parser.add_argument("--seed-only", action="store_true")
           parser.add_argument("--query-only", action="store_true")
           parser.add_argument("--full", action="store_true")
           parser.add_argument("--out-jsonl", type=Path, default=None)
           parser.add_argument("--backend-url", default="http://localhost:8080")
           parser.add_argument("--testclient", action="store_true")
           parser.add_argument("--seed-dir", type=Path, default=_DEFAULT_SEED_DIR)
           parser.add_argument("--questions", type=Path, default=_DEFAULT_QUESTIONS)
           parser.add_argument("--top-k", type=int, default=5)
           parser.add_argument("--floor", type=float, default=0.5,
                               help="Exit code 2 if recall@5 below this")
           args = parser.parse_args()

           if not (args.seed_only or args.query_only or args.full):
               parser.error("must specify one of --seed-only / --query-only / --full")

           client = _make_client(args.backend_url, args.testclient)
           try:
               if not _health_check(client):
                   print(json.dumps({"error": "backend unreachable"}), file=sys.stderr)
                   return 1

               if args.seed_only or args.full:
                   seed_result = _ingest_via_http(client, args.seed_dir)
                   if not args.full:
                       print(json.dumps(seed_result, ensure_ascii=False))
                       return 0

               if args.query_only or args.full:
                   questions = _load_questions(args.questions)
                   agg = _query_and_score(client, questions, top_k=args.top_k)
                   if args.out_jsonl:
                       with args.out_jsonl.open("w", encoding="utf-8") as f:
                           for row in agg["details"]:
                               f.write(json.dumps(row, ensure_ascii=False) + "\n")
                   summary = {k: v for k, v in agg.items() if k != "details"}
                   print(json.dumps(summary, ensure_ascii=False))
                   if agg["recall_at_5"] < args.floor:
                       return 2
                   return 0
           finally:
               if hasattr(client, "close"):
                   client.close()
           return 0


       if __name__ == "__main__":
           sys.exit(main())
       ```
