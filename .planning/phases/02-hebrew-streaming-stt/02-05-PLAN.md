```python
"""Hebrew-aware WER/CER computation (STT-05, RESEARCH §9).

Normalization: NFC + niqqud strip + punctuation strip + whitespace collapse.
Rationale: Hebrew reference transcripts often contain niqqud + cantillation
marks that the model never outputs; comparing without normalisation drives WER
artificially high. Standard practice for Hebrew ASR eval.
"""
from __future__ import annotations

import re
import unicodedata

import jiwer
from jiwer import Compose, RemoveMultipleSpaces, Strip

# \u0591-\u05C7 : Hebrew cantillation + niqqud
# \u200E-\u200F : LTR / RTL bidi control
# Plus standard punctuation stripped to spaces (preserves word boundaries).
_HEBREW_STRIP_RE = re.compile(
    r"[\u0591-\u05C7\u200E\u200F\.\,\!\?\:\;\"\'\(\)\[\]\-]"
)
_WHITESPACE_RE = re.compile(r"\s+")


def normalise_hebrew(text: str) -> str:
    """Canonical form for WER comparison.

    Pipeline: NFC normalize → strip niqqud + bidi + punctuation → collapse whitespace.
    """
    text = unicodedata.normalize("NFC", text)
    text = _HEBREW_STRIP_RE.sub(" ", text)
    text = _WHITESPACE_RE.sub(" ", text).strip()
    return text


_JIWER_TRANSFORM = Compose([Strip(), RemoveMultipleSpaces()])


def compute_wer(reference: str, hypothesis: str) -> dict[str, float]:
    """Return {'wer': float, 'cer': float} with Hebrew normalisation applied.

    Both metrics are unit-range [0.0, 1.0+]. WER counts word errors; CER counts
    character errors. For Hebrew, CER is often more robust (agglutinative
    morphology) so we report both.
    """
    r = normalise_hebrew(reference)
    h = normalise_hebrew(hypothesis)
    # jiwer 4.0 accepts str | Sequence[str]; empty strings are valid inputs.
    wer = jiwer.wer(
        r, h, reference_transform=_JIWER_TRANSFORM, hypothesis_transform=_JIWER_TRANSFORM
    )
    cer = jiwer.cer(
        r, h, reference_transform=_JIWER_TRANSFORM, hypothesis_transform=_JIWER_TRANSFORM
    )
    return {"wer": float(wer), "cer": float(cer)}
```
