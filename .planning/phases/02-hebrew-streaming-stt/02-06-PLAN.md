```python
"""Per-utterance STT metrics + loguru JSON sink (STT-06).

PII policy: the `text` field is PII. By default OMITTED from the log line;
still written to the SQLite audit table (filesystem-permissioned). Opt-in
override via settings.stt_log_text_redaction_disabled.
"""
from __future__ import annotations

import uuid
from dataclasses import dataclass
from datetime import datetime, timezone

from loguru import logger

from receptra.config import settings


@dataclass(frozen=True)
class UtteranceMetrics:
    utterance_id: str
    ts_utc: str
    t_speech_start_ms: int
    t_speech_end_ms: int
    t_final_ready_ms: int
    duration_ms: int
    transcribe_ms: int
    partials_emitted: int
    text: str                 # PII — redacted by default in log_utterance
    text_len_chars: int
    wer_sample_id: str | None = None

    @property
    def stt_latency_ms(self) -> int:
        """STT-06 metric. Clamped to >=0 to guard against monotonic clock edge cases."""
        return max(0, self.t_final_ready_ms - self.t_speech_end_ms)


def new_utterance_id() -> str:
    return uuid.uuid4().hex


def utc_now_iso() -> str:
    return datetime.now(tz=timezone.utc).isoformat()


def log_utterance(m: UtteranceMetrics) -> None:
    """Emit one structured JSON log line (event='stt.utterance')."""
    payload: dict[str, object] = {
        "utterance_id": m.utterance_id,
        "ts_utc": m.ts_utc,
        "t_speech_start_ms": m.t_speech_start_ms,
        "t_speech_end_ms": m.t_speech_end_ms,
        "t_final_ready_ms": m.t_final_ready_ms,
        "duration_ms": m.duration_ms,
        "stt_latency_ms": m.stt_latency_ms,
        "transcribe_ms": m.transcribe_ms,
        "partials_emitted": m.partials_emitted,
        "text_len_chars": m.text_len_chars,
        "wer_sample_id": m.wer_sample_id,
    }
    if settings.stt_log_text_redaction_disabled:
        payload["text"] = m.text
    logger.bind(event="stt.utterance").info(payload)
```

Create `backend/src/receptra/stt/audit.py`:

```python
"""SQLite audit log stub (Phase 2; Phase 5 owns final INT-05 schema).

Stdlib sqlite3. Lazy CREATE TABLE IF NOT EXISTS. sqlite3 connections are NOT
safe to share across threads; each call opens + closes within a `with` block.
"""
from __future__ import annotations

import sqlite3
from pathlib import Path

from receptra.stt.metrics import UtteranceMetrics

_SCHEMA = """
CREATE TABLE IF NOT EXISTS stt_utterances (
    utterance_id TEXT PRIMARY KEY,
    ts_utc TEXT NOT NULL,
    duration_ms INTEGER NOT NULL,
    stt_latency_ms INTEGER NOT NULL,
    transcribe_ms INTEGER NOT NULL,
    partials_emitted INTEGER NOT NULL,
    text TEXT NOT NULL,
    wer_sample_id TEXT
);
"""


def init_audit_db(path: str | Path) -> None:
    """Idempotent. Create parent dir + table if missing."""
    p = Path(path)
    p.parent.mkdir(parents=True, exist_ok=True)
    with sqlite3.connect(str(p)) as conn:
        conn.execute(_SCHEMA)
        conn.commit()


def insert_stt_utterance(path: str | Path, m: UtteranceMetrics) -> None:
    """Insert one row. Caller must have invoked init_audit_db before first use."""
    with sqlite3.connect(str(path)) as conn:
        conn.execute(
            """
            INSERT INTO stt_utterances
                (utterance_id, ts_utc, duration_ms, stt_latency_ms,
                 transcribe_ms, partials_emitted, text, wer_sample_id)
            VALUES (?, ?, ?, ?, ?, ?, ?, ?)
            """,
            (
                m.utterance_id,
                m.ts_utc,
                m.duration_ms,
                m.stt_latency_ms,
                m.transcribe_ms,
                m.partials_emitted,
                m.text,
                m.wer_sample_id,
            ),
        )
        conn.commit()
```

Step A — Edit `backend/src/receptra/stt/pipeline.py` (READ it first before editing per the Phase 1 pattern):

1. On connection accept: call `init_audit_db(settings.audit_db_path)` once.
2. Inside `run_utterance_loop`, track per-utterance state: `t_transcribe_start_ms` (before `await transcribe(...)` in the final path), `partials_emitted` counter, `utterance_id = new_utterance_id()` assigned at VAD start.
3. After sending the `FinalTranscript` event, build an `UtteranceMetrics(...)` and call BOTH `log_utterance(m)` AND `insert_stt_utterance(settings.audit_db_path, m)`. Do this in a try/except — SQLite failures MUST NOT crash the WebSocket loop. If insert fails, log a `loguru.error(event="stt.audit.insert_failed")` line with the utterance_id and the exception message (no PII).
4. In the `finally` cleanup path of `websocket_stt_endpoint`: if the disconnect happens during an active utterance, do NOT write a partial row to SQLite. Only completed utterances (reaching the final-emit code path) get audit rows. This is the contract the chaos test verifies.

Partial-emit counter: increment `partials_emitted += 1` every time a partial is sent; pass the final count into the final's `UtteranceMetrics`.

Pseudo-diff structure:
```
on VAD start:
    utterance_id = new_utterance_id()
    partials_emitted = 0
    t_speech_start_ms = ...
on partial emit:
    partials_emitted += 1
on VAD end:
    t_transcribe_start = monotonic()
    text = await transcribe(combined)
    t_final_ready_ms = monotonic_ms()
    transcribe_ms = (t_final_ready_ms - int(t_transcribe_start * 1000))
    send_json(FinalTranscript(...))
    m = UtteranceMetrics(
        utterance_id=utterance_id,
        ts_utc=utc_now_iso(),
        t_speech_start_ms=t_speech_start_ms,
        t_speech_end_ms=t_speech_end_ms,
        t_final_ready_ms=t_final_ready_ms,
        duration_ms=duration_ms,
        transcribe_ms=transcribe_ms,
        partials_emitted=partials_emitted,
        text=text,
        text_len_chars=len(text),
        wer_sample_id=None,
    )
    log_utterance(m)
    try:
        insert_stt_utterance(settings.audit_db_path, m)
    except Exception as e:
        logger.bind(event="stt.audit.insert_failed").error(
            {"utterance_id": utterance_id, "err": str(e)}
        )
```
