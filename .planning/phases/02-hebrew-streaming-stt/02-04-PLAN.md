```python
"""Pydantic event schema for /ws/stt (RESEARCH Â§8).

Discriminated union on `type` field. All timestamps are monotonic milliseconds.
Hebrew text is UTF-8 preserved; pydantic does not alter it.
"""
from __future__ import annotations

from typing import Annotated, Literal, Union

from pydantic import BaseModel, ConfigDict, Field


class SttReady(BaseModel):
    model_config = ConfigDict(frozen=True)
    type: Literal["ready"] = "ready"
    model: str
    sample_rate: int = 16000
    frame_bytes: int = 1024


class PartialTranscript(BaseModel):
    model_config = ConfigDict(frozen=True)
    type: Literal["partial"] = "partial"
    text: str
    t_speech_start_ms: int = Field(..., description="Monotonic ms when VAD start fired")
    t_emit_ms: int = Field(..., description="Monotonic ms when this event was built")


class FinalTranscript(BaseModel):
    model_config = ConfigDict(frozen=True)
    type: Literal["final"] = "final"
    text: str
    t_speech_start_ms: int
    t_speech_end_ms: int
    stt_latency_ms: int = Field(..., description="t_final_ready - t_speech_end (STT-06)")
    duration_ms: int = Field(..., description="Utterance audio duration in ms")


class SttError(BaseModel):
    model_config = ConfigDict(frozen=True)
    type: Literal["error"] = "error"
    code: Literal["model_error", "vad_error", "protocol_error"]
    message: str


SttEvent = Annotated[
    Union[SttReady, PartialTranscript, FinalTranscript, SttError],
    Field(discriminator="type"),
]
```

```python
"""WebSocket STT pipeline: VAD-gated incremental re-transcribe loop.

Wire contract (client â†’ server): binary frames, exactly 1024 bytes each,
int16 LE, 16 kHz mono, 32 ms per frame.
Wire contract (server â†’ client): JSON text frames per stt.events schema.

Concurrency: `model.transcribe()` is synchronous C++ and MUST be wrapped in
`asyncio.to_thread` (Pitfall #5). A single dedicated thread per connection
serialises that connection's transcribes; multiple connections run in
parallel threads.
"""
from __future__ import annotations

import asyncio
import time
from collections.abc import Awaitable, Callable

import numpy as np
from fastapi import WebSocket, WebSocketDisconnect
from loguru import logger

from receptra.config import settings
from receptra.stt.engine import transcribe_hebrew
from receptra.stt.events import FinalTranscript, PartialTranscript, SttError, SttReady
from receptra.stt.vad import FRAME_BYTES, InvalidFrameError, StreamingVad

MIN_PARTIAL_AUDIO_MS = 500  # RESEARCH Â§7 â€” don't transcribe < 500ms of audio


async def websocket_stt_endpoint(ws: WebSocket) -> None:
    """FastAPI route handler. Accepts, delegates, cleans up."""
    await ws.accept()
    whisper = ws.app.state.whisper
    vad_model = ws.app.state.vad_model
    model_name = getattr(whisper, "model_name", "ivrit-ai/whisper-large-v3-turbo-ct2")
    await ws.send_json(SttReady(model=model_name).model_dump())

    vad = StreamingVad(
        model=vad_model,
        threshold=settings.vad_threshold,
        min_silence_ms=settings.vad_min_silence_ms,
        speech_pad_ms=settings.vad_speech_pad_ms,
    )

    async def transcribe_call(buffer: np.ndarray[tuple[int], np.dtype[np.float32]]) -> str:
        text, _info = await asyncio.to_thread(transcribe_hebrew, whisper, buffer)
        return text

    try:
        await run_utterance_loop(ws, vad, transcribe_call)
    except WebSocketDisconnect:
        logger.bind(event="stt.ws.disconnect").info({"msg": "client disconnected mid-stream"})
    except Exception as e:  # pragma: no cover â€” defensive
        logger.bind(event="stt.ws.error").exception({"msg": "unexpected error", "err": str(e)})
        try:
            await ws.send_json(SttError(code="model_error", message=str(e)).model_dump())
        except Exception:
            pass
    finally:
        vad.reset()
        try:
            await ws.close()
        except Exception:
            pass


async def run_utterance_loop(
    ws: WebSocket,
    vad: StreamingVad,
    transcribe: Callable[[np.ndarray], Awaitable[str]],
) -> None:
    """Per-connection receive loop. Separated for testability and Plan 02-06 wrapping."""
    speech_buffer: list[np.ndarray] = []
    t_speech_start_ms = 0
    t_last_partial_ms = 0
    in_speech = False

    while True:
        frame = await ws.receive_bytes()
        try:
            event = vad.feed(frame)
        except InvalidFrameError as e:
            await ws.send_json(SttError(code="protocol_error", message=str(e)).model_dump())
            await ws.close(code=1007)
            return

        # Accumulate frame's float32 samples into the speech buffer if active
        if in_speech:
            pcm_f32 = np.frombuffer(frame, dtype="<i2").astype(np.float32) / 32768.0
            speech_buffer.append(pcm_f32)

        now_ms = int(time.monotonic() * 1000)

        if event and event["kind"] == "start":
            in_speech = True
            speech_buffer = []
            t_speech_start_ms = now_ms
            t_last_partial_ms = now_ms
            # Include the current frame as the first sample of the utterance
            pcm_f32 = np.frombuffer(frame, dtype="<i2").astype(np.float32) / 32768.0
            speech_buffer.append(pcm_f32)

        elif in_speech and event is None:
            # Maybe emit a partial
            audio_ms = (sum(s.shape[0] for s in speech_buffer) * 1000) // 16000
            elapsed_since_partial = now_ms - t_last_partial_ms
            if elapsed_since_partial >= settings.stt_partial_interval_ms and audio_ms >= MIN_PARTIAL_AUDIO_MS:
                combined = np.concatenate(speech_buffer)
                text = await transcribe(combined)
                await ws.send_json(
                    PartialTranscript(
                        text=text,
                        t_speech_start_ms=t_speech_start_ms,
                        t_emit_ms=int(time.monotonic() * 1000),
                    ).model_dump()
                )
                t_last_partial_ms = now_ms

        elif event and event["kind"] == "end":
            t_speech_end_ms = now_ms
            if not speech_buffer:
                in_speech = False
                continue
            combined = np.concatenate(speech_buffer)
            duration_ms = (combined.shape[0] * 1000) // 16000
            t_transcribe_start = time.monotonic()
            text = await transcribe(combined)
            t_final_ready_ms = int(time.monotonic() * 1000)
            stt_latency_ms = t_final_ready_ms - t_speech_end_ms
            await ws.send_json(
                FinalTranscript(
                    text=text,
                    t_speech_start_ms=t_speech_start_ms,
                    t_speech_end_ms=t_speech_end_ms,
                    stt_latency_ms=max(0, stt_latency_ms),
                    duration_ms=duration_ms,
                ).model_dump()
            )
            # Reset utterance state; VAD iterator keeps running across utterances
            speech_buffer = []
            in_speech = False
            t_speech_start_ms = 0
```
